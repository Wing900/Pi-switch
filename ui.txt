<script>
    function select(el, name, model) {
        document.querySelectorAll('.provider-card').forEach(c => c.classList.remove('active'));
        el.classList.add('active');
        document.getElementById('main-title').innerText = name;
        document.getElementById('opt-1').innerText = model;
        document.getElementById('opt-2').innerText = model + '-alt';
    }

    function openDrawer(name, event) {
        event.stopPropagation();
        document.getElementById('p-name').value = name;
        document.getElementById('drawer').classList.add('open');
    }

    function closeDrawer() {
        document.getElementById('drawer').classList.remove('open');
    }
</script>
